MySQL Schema

                        
-- Patients table
CREATE TABLE patients (
    patient_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    dob DATE,
    gender ENUM('male', 'female', 'other'),
    address TEXT,
    phone VARCHAR(20),
    email VARCHAR(100),
    registration_date DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Doctors table 
CREATE TABLE doctors (
    doctor_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    specialty VARCHAR(100),
    phone VARCHAR(20),
    email VARCHAR(100)
);

-- Appointments table
CREATE TABLE appointments (
    appointment_id INT AUTO_INCREMENT PRIMARY KEY,
    patient_id INT,
    doctor_id INT,
    appointment_date DATETIME,
    status ENUM('scheduled', 'confirmed', 'completed', 'cancelled'),
    notes TEXT,
    FOREIGN KEY (patient_id) REFERENCES patients(patient_id),
    FOREIGN KEY (doctor_id) REFERENCES doctors(doctor_id)
);

-- Get appointments for today
SELECT a.appointment_id, p.name AS patient_name, d.name AS doctor_name, 
       a.appointment_date, a.status

JOIN doctors d ON a.doctor_id = d.doctor_id
WHERE DATE(a.appointment_date) = CURDATE();

       b.date_issued, b.status
FROM bills b

-- Trigger for patient discharge
DELIMITER //
CREATE TRIGGER after_patient_discharge 
AFTER UPDATE ON patients
FOR EACH ROW
BEGIN
    IF NEW.status = 'discharged' THEN
        INSERT INTO billing (patient_id, amount, description)
        VALUES (NEW.patient_id, 5000.00, 'Hospital discharge fee');
    END IF;
END //
DELIMITER ;

Stored Procedures

                        
-- Calculate total bill
DELIMITER //
CREATE PROCEDURE CalculateTotalBill(IN patientId INT, OUT total DECIMAL(10,2))
BEGIN
    SELECT SUM(amount) INTO total 
    FROM bills 
    WHERE patient_id = patientId AND status != 'paid';
END //
DELIMITER ;

-- Generate monthly revenue report
DELIMITER //
CREATE PROCEDURE GenerateMonthlyRevenueReport(IN month INT, IN year INT)
BEGIN
    SELECT 
        p.name AS patient_name,
        b.bill_id,
        b.amount,
        b.date_issued,
        b.status
    FROM bills b
    JOIN patients p ON b.patient_id = p.patient_id
    WHERE MONTH(b.date_issued) = month AND YEAR(b.date_issued) = year
    ORDER BY b.date_issued ASC;
END //
DELIMITER ;
